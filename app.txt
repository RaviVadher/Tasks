For Leave Management system, create SP, Triggers and Views for: 
Write a function to return the number of leaves to be added if an employee joins anytime during the year, considering 1 day leave per month. 
Write a stored procedure such that as soon as a new Employee is added, his leave balance is added in Leave Balance table. Use the function created in previous step.
Create a stored procedure to cancel a Leave Request, update the leave balance of cancelled leave. and Rollback if any step fails.
Create a view to see the approved leave details for next 1 month for all employees.
Create a view to have Department-wise, employee-wise total number of all leaves. 
Write a trigger such that when a new Leave Type is added, its related Leave details and balance are added as well. 
Write a trigger such that as soon as a Leave Request is approved, the remaining balance should get updated.


--departments 
CREATE TABLE departments (
    pk_dept_id INT PRIMARY KEY IDENTITY(1,1) ,
    dept_name VARCHAR(100) NOT NULL
);

--employees
CREATE TABLE employees (
    pk_emp_id INT PRIMARY KEY IDENTITY(1,1),
    emp_name NVARCHAR(100) NOT NULL,
    fk_dept_emp INT REFERENCES departments(pk_dept_id),
    joining_date DATE NOT NULL
);

-- leave_types
CREATE TABLE leave_types (
    pk_types_id INT PRIMARY KEY IDENTITY(1,1),
    leave_type_name NVARCHAR(50) NOT NULL,
    total INT NOT NULL
);

-- leave_application_table
CREATE TABLE leave_applications (
    pk_app_id INT PRIMARY KEY IDENTITY(1,1),
    fk_app_emp INT FOREIGN KEY REFERENCES employees(pk_emp_id),
    fk_app_type INT FOREIGN KEY REFERENCES  leave_types(pk_types_id),
    leave_date DATE NOT NULL,
    approvedby NVARCHAR(150),
    Status NVARCHAR(20) CHECK (Status IN ('Approved','Rejected'))
);

-- Leav_balances
CREATE TABLE leave_balances (
    fk_balance_emp INT FOREIGN KEY REFERENCES employees(pk_emp_id),
    fk_balance_type INT FOREIGN KEY REFERENCES  leave_types(pk_types_id),
    Year INT NOT NULL,
    leave_taken INT DEFAULT 0,
    PRIMARY KEY (fk_balance_emp, fk_balance_type, Year)
);

    e.emp_name AS EmployeeName
FROM leave_applications la
JOIN employees e 
    ON la.fk_app_emp = e.pk_emp_id

AND la.Status = 'Approved';

SELECT 
    lt.leave_type_name,
    COUNT(*) AS TotalLeaves
FROM leave_applications la
JOIN leave_types lt 
    ON la.fk_app_type = lt.pk_types_id
WHERE MONTH(la.leave_date) = MONTH(GETDATE())
AND YEAR(la.leave_date) = YEAR(GETDATE())
AND la.Status = 'Approved'
GROUP BY lt.leave_type_name;

CREATE TRIGGER trg_restore_balance_on_leave_cancel
ON leave_applications
AFTER UPDATE
AS
BEGIN
    UPDATE lb
    SET lb.leave_taken = lb.leave_taken - 1
    FROM leave_balances lb
    JOIN deleted d
        ON lb.fk_balance_emp = d.fk_app_emp
        AND lb.fk_balance_type = d.fk_app_type
        AND lb.Year = YEAR(d.leave_date)
    JOIN inserted i
        ON i.pk_app_id = d.pk_app_id
    WHERE d.Status = 'Approved'
      AND i.Status = 'Rejected'
      AND lb.leave_taken > 0;
END;
